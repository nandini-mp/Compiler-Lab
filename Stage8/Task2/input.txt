class
A
{
decl
    int l;
    int b;
    int dim();
    int findArea();
enddecl
int dim() {
    decl
    enddecl
    begin
        write("Enter l");
        read(self.l);
        write("Enter b");
        read(self.b);
        return 1;
    end
}
int findArea() {
    decl
    enddecl
    begin
        return self.l*self.b;
    end
}
}
B extends A
{
    decl
    int dim();
    enddecl
    int dim() {
        decl
        enddecl
        begin
            write("Enter s");
            read(self.l);
            self.b = self.l;
            return 1;
        end
    }
}
endclass

decl
    A obj;
enddecl

int main() {
    decl
        int n;
    enddecl
    begin
        n=initialize();
        write("1 for r, 2 for s");
        read(n);
        if(n==1) then
            write("Going to create A");
            obj = new(A);
            write("A created");
        else
            obj = new(B);
        endif;
        n = obj.dim();
        write(obj.findArea());
        return 0;
    end
}
